db.jugadores.aggregate([
  {
    "$lookup": {
      "from": "stats",
      "localField": "codigo",
      "foreignField": "jugador",
      "as": "puntos"
    }
  },
  {"$match : {"puntos.temporada": "temporada_deseada"}},
  {
    "$project": {
      "nombre": 1,
      "puntos.PPP": 1
    }
  },
  {
    "$sort": {
      "puntos.PPP": -1
    }
  }
])
